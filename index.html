// 覆蓋掉你原本的 onValue 區塊，這是最穩定的讀取路徑 onValue(dataRef, (snapshot) => { const cloudData = snapshot.val(); if (cloudData) { window.appData = cloudData; window.dynamicPanels = []; // 核心修正：強制遍歷所有資料夾，不管名稱為何 for (let key in cloudData) { if (['timeline', 'seating', 'checklist', 'tablesConfig', 'timelineConfig'].includes(key)) continue; // 只要該節點有 name，就當作一個面板，不管它是什麼名字 if (cloudData[key] && typeof cloudData[key] === 'object') { window.dynamicPanels.push({ id: key, name: cloudData[key].name || key, // 如果沒名稱，就顯示 ID icon: cloudData[key].icon || "fa-folder" }); } } window.renderSidebarMenu(); window.renderActivePanelOnly(); } }); // 修正渲染邏輯，確保資料沒對齊時不會顯示空白 window.renderActivePanelOnly = function() { if (!window.appData) return; const tabId = window.currentActiveTab || "timeline"; document.querySelectorAll('.main-panel').forEach(p => p.classList.add('hidden')); if(tabId === 'timeline') { document.getElementById('panel-timeline').classList.remove('hidden'); renderTimelineTable(); } else if(tabId === 'seating') { document.getElementById('panel-seating').classList.remove('hidden'); renderVenueTablesGrid(); } else if(tabId === 'checklist') { document.getElementById('panel-checklist').classList.remove('hidden'); renderChecklist(); } else { // 這裡：直接讀取節點，如果結構不同，它會自動抓取裡面的 members 和 modules const teamData = window.appData[tabId] || {}; document.getElementById('panel-team-generic').classList.remove('hidden'); document.getElementById('team-title-display').innerText = teamData.name || tabId; const membersBox = document.getElementById('team-members-box'); membersBox.innerText = teamData.members || "尚無成員紀錄..."; membersBox.onblur = function() { if(window.cloudUpdateCell) window.cloudUpdateCell(`${tabId}/members`, this.innerText); }; const delBtn = document.getElementById('btn-delete-entire-panel'); delBtn.onclick = function() { window.deleteEntireCustomPanel(tabId); }; renderTeamModules(teamData, tabId); } }
